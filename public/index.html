async function updateMasterCSV() { const fileInput = document.createElement('input'); fileInput.type = 'file'; fileInput.accept = '.csv'; fileInput.onchange = async () => { const file = fileInput.files[0]; if (!file) return; const formData = new FormData(); formData.append('csvFile', file); // This key must match upload.single('csvFile') const btn = document.querySelector('button[onclick="updateMasterCSV()"]'); const originalText = btn.innerText; btn.innerText = "UPLOADING..."; btn.disabled = true; try { const res = await fetch('/api/upload-csv', { method: 'POST', body: formData }); if (res.ok) { alert("Master Inventory Updated Successfully!"); } else { const errText = await res.text(); alert("Upload Failed: " + errText); } } catch (err) { alert("Error: " + err.message); } finally { btn.innerText = originalText; btn.disabled = false; } }; fileInput.click(); }
